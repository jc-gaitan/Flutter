
abstract class Vehiculo{
  String? color;
  String? modelo;
  String? marca;
  
  //metodo abstracto
  void miCombustible();
  
}

//interface --> con implemets se usa la interface
abstract class EsTransporte{
  void queTransporteSoy();
}

class Camion extends Vehiculo implements EsTransporte{
  @override
  void miCombustible(){
    print("Diesel");
  }
  
  @override
  void queTransporteSoy(){
    print("Soy un camion");
  }
  
}

class Auto extends Vehiculo{
  @override
  void miCombustible(){
    print("Gasolina");
  }
  
}

class Moto extends Vehiculo{
  @override
  void miCombustible(){
    print("Gasolina");
  }
  
}

void main() {

  var camion = new Camion();
  camion.color = "Rojo";
  print(camion.color);
  camion.miCombustible();
  camion.queTransporteSoy();
  
}
