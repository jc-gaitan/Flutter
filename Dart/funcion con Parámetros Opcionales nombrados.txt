//funcion con Par√°metros Opcionales nombrados
void saludo([String nombre = "", String apellido = ""]){
  var saludos = new StringBuffer("Bienvenido a Dart");
  if(nombre !=null){
    saludos.write(" $nombre");
  }
  if(apellido!=null){
    saludos.write(" $apellido");
  }
  
  print(saludos.toString());
}
